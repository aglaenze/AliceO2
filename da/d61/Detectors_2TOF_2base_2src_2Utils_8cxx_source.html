51 double Utils::subtractInteractionBC(double time)
52 {
53  int bc = int(time * o2::tof::Geo::BC_TIME_INPS_INV + 0.2);
54  int bcOrbit = bc % o2::constants::lhc::LHCMaxBunches;
55 
56  int dbc = o2::constants::lhc::LHCMaxBunches, bcc = bc;
57  for (int k = 0; k < getNinteractionBC(); k++) { // get bc from fill scheme closest
58  if (abs(bcOrbit - getInteractionBC(k)) < dbc) {
59  bcc = bc - bcOrbit + getInteractionBC(k);
60  dbc = abs(bcOrbit - getInteractionBC(k));
61  }
62  if (abs(bcOrbit - getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches) < dbc) { // in case k is close to the right border (last BC of the orbit)
63  bcc = bc - bcOrbit + getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches;
64  dbc = abs(bcOrbit - getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches);
65  }
66  if (abs(bcOrbit - getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches) < dbc) { // in case k is close to the left border (BC=0)
67  bcc = bc - bcOrbit + getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches;
68  dbc = abs(bcOrbit - getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches);
69  }
70  }
71  time -= o2::tof::Geo::BC_TIME_INPS * bcc;
72 
73  return time;
74 }

93 void Utils::addBC(float toftime)
94 {
95  if (!mIsInit) {
96  init();
97  mIsInit = true;
98  }
99 
100  if (mNautodet > MAX_NUM_EVENT_AUTODETECT) {
101  if (!hasFillScheme()) { // detect fill scheme
102  int thres = mMaxBC / 2;
103  for (int i = 0; i < o2::constants::lhc::LHCMaxBunches; i++) {
104  if (mBCmult[i] > thres) { // good bunch
105  addInteractionBC(i);
106  }
107  }
108  }
109  return;
110  }
111 
112  // just fill
113  int bc = int(toftime * o2::tof::Geo::BC_TIME_INPS_INV + 0.2) % o2::constants::lhc::LHCMaxBunches;
114 
115  mBCmult[bc]++;
116 
117  if (mBCmult[bc] > mMaxBC) {
118  mMaxBC = mBCmult[bc];
119  }
120  mNautodet++;
121 }
